60 bool readLineFromData(const QByteArray &data, int &dataPos, int &lineStart, int &lineLen, int &equalsPos)
61 {
62  int dataLen = data.length();
63 
64  equalsPos = -1;
65 
66  lineStart = dataPos;
67  while (lineStart < dataLen && (charTraits[uint(uchar(data.at(lineStart)))] & Space))
68  ++lineStart;
69 
70  int i = lineStart;
71  while (i < dataLen) {
72  while (!(charTraits[uint(uchar(data.at(i)))] & Special)) {
73  if (++i == dataLen)
74  goto break_out_of_outer_loop;
75  }
76 
77  char ch = data.at(i++);
78  if (ch == '=') {
79  if (equalsPos == -1)
80  equalsPos = i - 1;
81  } else if (ch == '\n' || ch == '\r') {
82  if (i == lineStart + 1) {
83  ++lineStart;
84  } else {
85  --i;
86  goto break_out_of_outer_loop;
87  }
88  } else if (ch == '\\') {
89  if (i < dataLen) {
90  char ch = data.at(i++);
91  if (i < dataLen) {
92  char ch2 = data.at(i);
93  // \n, \r, \r\n, and \n\r are legitimate line terminators in INI files
94  if ((ch == '\n' && ch2 == '\r') || (ch == '\r' && ch2 == '\n'))
95  ++i;
96  }
97  }
98  } else if (ch == ';') {
99  // The multiple values should be separated by a semicolon and the value of the key
100  // may be optionally terminated by a semicolon. Trailing empty strings must always
101  // be terminated with a semicolon. Semicolons in these values need to be escaped
102  // using \; .
103  // Don't need to do anything here.
104  } else {
105  Q_ASSERT(ch == '#');
106 
107  if (i == lineStart + 1) {
108  char ch;
109  while (i < dataLen && (((ch = data.at(i)) != '\n') && ch != '\r'))
110  ++i;
111  lineStart = i;
112  }
113  }
114  }
115 
116 break_out_of_outer_loop:
117  dataPos = i;
118  lineLen = i - lineStart;
119  return lineLen > 0;
120 }

122 QString &doEscape(QString& str, const QHash<QChar,QChar> &repl)
123 {
124  // First we replace slash.
125  str.replace(QLatin1Char('\\'), QLatin1String("\\\\"));
126 
127  QHashIterator<QChar,QChar> i(repl);
128  while (i.hasNext()) {
129  i.next();
130  if (i.key() != QLatin1Char('\\'))
131  str.replace(i.key(), QString::fromLatin1("\\\\%1").arg(i.value()));
132  }
133 
134  return str;
135 }

137 QString &doUnescape(QString& str, const QHash<QChar,QChar> &repl)
138 {
139  int n = 0;
140  while (1) {
141  n=str.indexOf(QLatin1String("\\"), n);
142  if (n < 0 || n > str.length() - 2)
143  break;
144 
145  if (repl.contains(str.at(n+1))) {
146  str.replace(n, 2, repl.value(str.at(n+1)));
147  }
148 
149  n++;
150  }
151 
152  return str;
153 }

156 class DDesktopEntrySection

158 public:

161  QString name;
162  QMap<QString, QString> valuesMap;
163  QByteArray unparsedDatas;

166  inline operator QString() const {
167  return QLatin1String("DDesktopEntrySection(") + name + QLatin1String(")");

170  QByteArray sectionData() const {
171  if (unparsedDatas.isEmpty()) {
172  // construct data and return
173  QByteArray data;
174 
175  data.append(QString("[%1]\n").arg(name));
176 
177  QMap<QString, QString>::const_iterator i;
178  for (i = valuesMap.begin(); i != valuesMap.end(); i++) {
179  data.append(QString("%1=%2\n").arg(i.key(), i.value()));
180  }
181 
182  return data;
183  } else {
184  return unparsedDatas;
185  }
186  }
187 
188  bool ensureSectionDataParsed() {
189  if (unparsedDatas.isEmpty()) return true;
190 
191  valuesMap.clear();
192 
193  // for readLineFromFileData()
194  int dataPos = 0;
195  int lineStart;
196  int lineLen;
197  int equalsPos;
198 
199  while(readLineFromData(unparsedDatas, dataPos, lineStart, lineLen, equalsPos)) {
200  if (unparsedDatas.at(lineStart) == '[') continue; // section name already parsed
201 
202  if (equalsPos != -1) {
203  QString key = unparsedDatas.mid(lineStart, equalsPos - lineStart).trimmed();
204  QString rawValue = unparsedDatas.mid(equalsPos + 1, lineStart + lineLen - equalsPos - 1).trimmed();
205 
206  valuesMap[key] = rawValue;
207  }
208  }
209 
210  unparsedDatas.clear();
211 
212  return true;
213  }
214 
215  bool contains(const QString &key) const {
216  const_cast<DDesktopEntrySection*>(this)->ensureSectionDataParsed();
217  return valuesMap.contains(key);
218  }
219 
220  QStringList allKeys() const {
221  const_cast<DDesktopEntrySection*>(this)->ensureSectionDataParsed();
222  return valuesMap.keys();
223  }
224 
225  QString get(const QString &key, QString &defaultValue) {
226  if (this->contains(key)) {
227  return valuesMap[key];
228  } else {
229  return defaultValue;
230  }
231  }
232 
233  bool set(const QString &key, const QString &value) {
234  if (this->contains(key)) {
235  valuesMap.remove(key);
236  }
237  valuesMap[key] = value;
238  return true;
239  }
240 
241  bool remove(const QString &key) {
242  if (this->contains(key)) {
243  valuesMap.remove(key);
244  return true;
245  }
246  return false;
247  }

345 bool DDesktopEntryPrivate::initSectionsFromData(const QByteArray &data)
346 {
347  sectionsMap.clear();
348 
349  QString lastSectionName;
350  int lastSectionStart = 0;
351  bool formatOk = true;
352  int sectionIdx = 0;
353  // for readLineFromFileData()
354  int dataPos = 0;
355  int lineStart;
356  int lineLen;
357  int equalsPos;
358 
359  auto commitSection = [=](const QString &name, int sectionStartPos, int sectionLength, int sectionIndex) {
360  DDesktopEntrySection lastSection;
361  lastSection.name = name;
362  lastSection.unparsedDatas = data.mid(sectionStartPos, sectionLength);
363  lastSection.sectionPos = sectionIndex;
364  sectionsMap[name] = lastSection;
365  };
366 
367  // TODO: here we only need to find the section start, so things like equalsPos are useless here.
368  // maybe we can do some optimization here via adding extra argument to readLineFromData().
369  while(readLineFromData(data, dataPos, lineStart, lineLen, equalsPos)) {
370  // qDebug() << "CurrentLine:" << data.mid(lineStart, lineLen);
371  if (data.at(lineStart) == '[') {
372  // commit the last section we've ever read before we read the new one.
373  if (!lastSectionName.isEmpty()) {
374  commitSection(lastSectionName, lastSectionStart, lineStart - lastSectionStart, sectionIdx);
375  sectionIdx++;
376  }
377  // process section name line
378  QByteArray sectionName;
379  int idx = data.indexOf(']', lineStart);
380  if (idx == -1 || idx >= lineStart + lineLen) {
381  qWarning() << "Bad desktop file format while reading line:" << data.mid(lineStart, lineLen);
382  formatOk = false;
383  sectionName = data.mid(lineStart + 1, lineLen - 1).trimmed();
384  } else {
385  sectionName = data.mid(lineStart + 1, idx - lineStart - 1).trimmed();
386  }
387  lastSectionName = sectionName;
388  lastSectionStart = lineStart;
389  }
390  }
391 
392  Q_ASSERT(lineStart == data.length());
393  if (!lastSectionName.isEmpty()) {
394  commitSection(lastSectionName, lastSectionStart, lineStart - lastSectionStart, sectionIdx);
395  }
396 
397  return formatOk;
398 }

753 QString DDesktopEntry::localizedValue(const QString &key, const QString &localeKey, const QString &section, const QString &defaultValue) const
754 {
755  Q_D(const DDesktopEntry);
756  QString result = defaultValue;
757  QString actualLocaleKey = QLatin1String("C");
758  if (key.isEmpty() || section.isEmpty()) {
759  qWarning("DDesktopEntry::localizedValue: Empty key or section passed");
760  return result;
761  }
762 
763  QStringList possibleKeys;
764 
765  // 此处添加 bcp47Name() 是为了兼容 desktop 文件中的语言长短名解析。
766  // 比如芬兰语，有 [fi] 和 [fi_FI] 两种情况，QLocale::name() 对应 fi_FI，QLocale::bcp47Name() 对应 fi。
767  if (!localeKey.isEmpty()) {
768  if (localeKey == "empty") {
769  possibleKeys << key;
770  } else if (localeKey == "default") {
771  possibleKeys << QString("%1[%2]").arg(key, QLocale().name());
772  possibleKeys << QString("%1[%2]").arg(key, QLocale().bcp47Name());
773  } else if (localeKey == "system") {
774  possibleKeys << QString("%1[%2]").arg(key, QLocale::system().name());
775  possibleKeys << QString("%1[%2]").arg(key, QLocale::system().bcp47Name());
776  } else {
777  possibleKeys << QString("%1[%2]").arg(key, localeKey);
778  }
779  }
780 
781  if (!actualLocaleKey.isEmpty()) {
782  possibleKeys << QString("%1[%2]").arg(key, actualLocaleKey);
783  }
784  possibleKeys << QString("%1[%2]").arg(key, "C");
785  possibleKeys << key;
786 
787  for (const QString &oneKey : possibleKeys) {
788  if (d->contains(section, oneKey)) {
789  const_cast<DDesktopEntryPrivate *>(d)->get(section, oneKey, &result);
790  break;
791  }
792  }
793 
794  return result;
795 }
