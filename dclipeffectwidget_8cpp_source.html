232 void DClipEffectWidget::paintEvent(QPaintEvent *event)
233 {
234  D_DC(DClipEffectWidget);
235 
236  if (d->image.isNull())
237  return;
238 
239  qreal devicePixelRatio = devicePixelRatioF();
240  const QRectF &rect = QRectF(event->rect()) & QRectF(this->rect()).marginsRemoved(d->margins);
241  const QPoint &offset = mapTo(window(), QPoint(0, 0));
242  const QRectF &imageRect = multiply(rect, devicePixelRatio) & d->imageGeometry.translated(-offset * devicePixelRatio);
243 
244  if (!imageRect.isValid())
245  return;
246 
247  QPainter p(this);
248  QPainterPath newPath;
249 
250  newPath.addRect(this->rect());
251  newPath -= d->path;
252 
253  p.setRenderHint(QPainter::Antialiasing);
254  p.setClipPath(newPath);
255  p.setCompositionMode(QPainter::CompositionMode_Source);
256  p.drawImage(imageRect.topLeft() / devicePixelRatio, d->image, imageRect);
257 }
