184 static bool tryAcquireSystemSemaphore(QSystemSemaphore *ss, qint64 timeout = 10)
185 {
186  if (ss->error() != QSystemSemaphore::NoError) {
187  return false;
188  }
189 
190  QSystemSemaphore _tmp_ss(QString("%1-%2").arg("DTK::tryAcquireSystemSemaphore").arg(ss->key()), 1, QSystemSemaphore::Open);
191 
192  _tmp_ss.acquire();
193 
194  QElapsedTimer t;
195  QFuture<bool> request = QtConcurrent::run(ss, &QSystemSemaphore::acquire);
196 
197  t.start();
198 
199  while (Q_LIKELY(t.elapsed() < timeout && !request.isFinished()));
200 
201  if (request.isFinished()) {
202  return true;
203  }
204 
205  if (Q_LIKELY(request.isRunning())) {
206  if (Q_LIKELY(ss->release(1))) {
207  request.waitForFinished();
208  }
209  }
210 
211  return false;
212 }

214 bool DApplicationPrivate::setSingleInstanceBySemaphore(const QString &key)
215 {
216  static QSystemSemaphore ss(key, 1, QSystemSemaphore::Open);
217  static bool singleInstance = false;
218 
219  if (singleInstance) {
220  return true;
221  }
222 
223  Q_ASSERT_X(ss.error() == QSystemSemaphore::NoError, "DApplicationPrivate::setSingleInstanceBySemaphore:", ss.errorString().toLocal8Bit().constData());
224 
225  singleInstance = tryAcquireSystemSemaphore(&ss);
226 
227  if (singleInstance) {
228  QtConcurrent::run([this] {
229  QPointer<DApplication> that = q_func();
230 
231  while (ss.acquire() && singleInstance)
232  {
233  if (!that) {
234  return;
235  }
236 
237  if (that->startingUp() || that->closingDown()) {
238  break;
239  }
240 
241  ss.release(1);
242 
243  if (that) {
244  Q_EMIT that->newInstanceStarted();
245  }
246  }
247  });
248 
249  auto clean_fun = [] {
250  ss.release(1);
251  singleInstance = false;
252  };
253 
254  qAddPostRoutine(clean_fun);
255  atexit(clean_fun);
256  }
257 
258  return singleInstance;
259 }

317 bool DApplicationPrivate::loadTranslator(QList<DPathBuf> translateDirs, const QString &name, QList<QLocale> localeFallback)
318 {
319  D_Q(DApplication);
320 
321  QStringList missingQmfiles;
322  for (auto &locale : localeFallback) {
323  QString translateFilename = QString("%1_%2").arg(name).arg(locale.name());
324  for (auto &path : translateDirs) {
325  QString translatePath = (path / translateFilename).toString();
326  if (QFile::exists(translatePath + ".qm")) {
327  qDebug() << "load translate" << translatePath;
328  auto translator = new QTranslator(q);
329  translator->load(translatePath);
330  q->installTranslator(translator);
331  return true;
332  }
333  }
334 
335  // fix english does not need to translation..
336  if (locale.language() != QLocale::English) {
337  missingQmfiles << translateFilename + ".qm";
338  }
339 
340  QStringList parseLocalNameList = locale.name().split("_", QString::SkipEmptyParts);
341  if (parseLocalNameList.length() > 0) {
342  translateFilename = QString("%1_%2").arg(name)
343  .arg(parseLocalNameList.at(0));
344  for (auto &path : translateDirs) {
345  QString translatePath = (path / translateFilename).toString();
346  if (QFile::exists(translatePath + ".qm")) {
347  qDebug() << "translatePath after feedback:" << translatePath;
348  auto translator = new QTranslator(q);
349  translator->load(translatePath);
350  q->installTranslator(translator);
351  return true;
352  }
353  }
354  }
355 
356  // fix english does not need to translation..
357  if (locale.language() != QLocale::English) {
358  missingQmfiles << translateFilename + ".qm";
359  }
360  }
361 
362  if (missingQmfiles.size() > 0) {
363  qWarning() << name << "can not find qm files" << missingQmfiles;
364  }
365  return false;
366 }
