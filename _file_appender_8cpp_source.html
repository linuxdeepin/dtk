72 bool FileAppender::openFile()
73 {
74  bool isOpen = m_logFile.isOpen();
75  if (!isOpen)
76  {
77  isOpen = m_logFile.open(QIODevice::WriteOnly | QIODevice::Append | QIODevice::Text);
78  if (isOpen)
79  m_logStream.setDevice(&m_logFile);
80  else
81  std::cerr << "<FileAppender::append> Cannot open the log file " << qPrintable(m_logFile.fileName()) << std::endl;
82  }
83  return isOpen;
84 }
