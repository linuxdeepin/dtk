236 void DInstanceGuard::destroy()
237 {
238  if (!s_pSelf) {
239  return;
240  }
241 
242  bool shmRelease = false;
243  int anotherProc = !s_nLock;
244  if (!QFile(s_procIdPath[anotherProc]).exists()) {
245  shmRelease = true;
246  pthread_mutex_unlock(&s_pShm->mutex[anotherProc]);
247  }
248  pthread_mutex_unlock(&s_pShm->mutex[s_nLock]);
249  if (s_pShm->CriticalSection.criticalProcessPid == getpid()) {
250  s_pShm->CriticalSection.criticalProcessPid = 0;
251  pthread_mutex_unlock(&s_pShm->CriticalSection.criticalSectionMtx);
252  }
253  if (shmRelease) {
254  shmctl(s_shmId, IPC_RMID, nullptr);
255  }
256 
257  delete s_pSelf;
258  s_pSelf = nullptr;
259 }

626 QColor DGuiApplicationHelper::blendColor(const QColor &substrate, const QColor &superstratum)
627 {
628  QColor c2 = superstratum.toRgb();
629 
630  if (c2.alpha() >= 255)
631  return c2;
632
