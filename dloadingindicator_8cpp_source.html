33 void DLoadingIndicatorPrivate::init()
34 {
35  D_Q(DLoadingIndicator);
36 
37  q->setSizePolicy(QSizePolicy::Fixed, QSizePolicy::Fixed);
38  q->setScene(new QGraphicsScene(q));
39  q->setRenderHint(QPainter::SmoothPixmapTransform);
40  q->setViewportUpdateMode(QGraphicsView::SmartViewportUpdate);
41  q->setHorizontalScrollBarPolicy(Qt::ScrollBarAlwaysOff);
42  q->setVerticalScrollBarPolicy(Qt::ScrollBarAlwaysOff);
43  q->viewport()->setAccessibleName("DLoadingIndicatorViewport");
44 
45  rotateAni.setDuration(1000);
46  rotateAni.setEasingCurve(QEasingCurve::OutInQuad);
47 
48  rotateAni.setLoopCount(-1);
49  rotateAni.setStartValue(QVariant(qreal(0.0)));
50  rotateAni.setEndValue(QVariant(qreal(360.0)));
51 
52  q->connect(&rotateAni, SIGNAL(valueChanged(QVariant)), q, SLOT(setRotate(QVariant)));
53 }

191 void DLoadingIndicator::setSmooth(bool smooth)
192 {
193  D_D(DLoadingIndicator);
194 
195  if(d->smooth == smooth)
196  return;
197 
198  d->smooth = smooth;
199 
200  QGraphicsPixmapItem * item = nullptr;
201 
202  if(!scene()->items().isEmpty())
203  item = dynamic_cast<QGraphicsPixmapItem*>(scene()->items().first());
204 
205  if(smooth){
206  setRenderHints(QPainter::SmoothPixmapTransform | QPainter::Antialiasing);
207  if(item)
208  item->setTransformationMode(Qt::SmoothTransformation);
209  }else{
210  setRenderHint(QPainter::SmoothPixmapTransform, false);
211  setRenderHint(QPainter::Antialiasing, false);
212  if(item)
213  item->setTransformationMode(Qt::FastTransformation);
214  }
215 }

217 void DLoadingIndicator::setDirection(DLoadingIndicator::RotationDirection direction)
218 {
219  D_D(DLoadingIndicator);
220 
221  if (d->direction == direction)
222  return;
223 
224  d->direction = direction;
225 
226  if(direction == Clockwise) {
227  d->rotateAni.setStartValue(QVariant(qreal(0.0)));
228  d->rotateAni.setEndValue(QVariant(qreal(360.0)));
229  } else {
230  d->rotateAni.setStartValue(QVariant(qreal(0.0)));
231  d->rotateAni.setEndValue(QVariant(qreal(-360.0)));
232  }
233 
234  Q_EMIT directionChanged(direction);
235 }
